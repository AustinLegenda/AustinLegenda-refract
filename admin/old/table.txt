    id INT AUTO_INCREMENT PRIMARY KEY,
    year INT NOT NULL,
    month INT NOT NULL,
    day INT NOT NULL,
    hour INT NOT NULL,
    minute INT NOT NULL,
    wvht FLOAT NOT NULL,
    swh FLOAT NOT NULL,
    swp FLOAT NOT NULL,
    wwh FLOAT NOT NULL,
    wwp FLOAT NOT NULL,
    swd FLOAT NOT NULL,
    wwd FLOAT NOT NULL,
    steepness FLOAT NOT NULL,
    apd FLOAT NOT NULL,
    mwd FLOAT NOT NULL
);
